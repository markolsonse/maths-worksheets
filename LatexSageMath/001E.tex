\begin{sagesilent}

t=ZZ.random_element(999999)
set_random_seed(t)

a=[(ZZ.random_element(1,26)) for i in range(36)]
b=[(ZZ.random_element(1,26)) for i in range(len(a))]
c=[(ZZ.random_element(1,26)) for i in range(len(a))]
d=[(ZZ.random_element(1,26)) for i in range(len(a))]

for j in range(len(c)):
  while (c[j]-a[j])==0:
    c[j]=(-1)^(ZZ.random_element(1,3))*(ZZ.random_element(1,26))
    a[j]=ZZ.random_element(1,26)

answer=[(d[i]-b[i])/(c[i]-a[i]) for i in range(len(a))]

qoutput=""
aoutput=""

for i in range(len(a)):
  qoutput += r"\item $P_1(%s, %s)$ , $P_2(%s, %s)$" %(a[i],b[i],c[i],d[i])
  aoutput += r"\item \cRed{$m=%s$}" %(answer[i])

\end{sagesilent}
