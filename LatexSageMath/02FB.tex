\begin{sagesilent}
# Generate and set a random seed.
t=ZZ.random_element(999999)
set_random_seed(t)

# Generate lists and assign variables
a=[ZZ.random_element(1,13) for i in range(49)]
b=[ZZ.random_element(1,13)  for i in range(len(a))]
c=[ZZ.random_element(1,13)  for i in range(len(a))]
d=[ZZ.random_element(1,13)  for i in range(len(a))]

# Generate the questions & answers
numerator=[latex(a[i]*b[i]) for i in range(len(a))]
denominator=[latex(c[i]*d[i]) for i in range(len(a))]
fnumerator=[latex(factor(a[i]*b[i])) for i in range(len(a))]
fdenominator=[latex(factor(c[i]*d[i])) for i in range(len(a))]
answera=[latex((a[i]*b[i])/(c[i]*d[i])) for i in range(len(a))]

qoutput=""
aoutput=""


for i in range(len(b)):
  qoutput += f"\item \( \dfrac{{ {numerator[i]} }}{{ {denominator[i]} }} \)" 
  aoutput += f"\item \cRed{{ \( \dfrac{{ {fnumerator[i]} }}{{ {fdenominator[i]} }} = {answera[i]} \)}}" 

\end{sagesilent}
